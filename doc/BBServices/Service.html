def run(&block)
  set_ran
  begin
    initialize_service
    run_service
  rescue => e
    set_successful(false)
    set_error(e)
  ensure
    call_block(&block)
  end
end

def run!(&block)
  set_ran
  begin
    initialize_service
    run_service!
    call_block(&block)
  rescue => e
    set_successful(false)
    set_error(e)
    raise e
  end
end
