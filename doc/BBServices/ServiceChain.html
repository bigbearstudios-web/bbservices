def chain(params = {})
  tap do |_service_chain|
    if @successful
      service = yield(params, self, previous_service)
      process_service(service)
    end
  end
end

def on(success: proc {}, failure: proc {})
  if successful?
    success.call
  else
    failure.call
  end
end
